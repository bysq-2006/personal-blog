## 基于Compressor
[Compressor](https://github.com/fengyuanchen/compressorjs) 是一个基于 JavaScript 的图片压缩工具，支持在浏览器端对图片进行压缩处理。它可以有效地减小图片文件的大小，同时保持较高的图像质量，适用于各种 web 应用场景。

## 工具本身

<div class="main">
  <div class="upload-box"
        @click="triggerFileInput"
        @dragover.prevent
        @drop.prevent="handleDrop">
    <span v-if="!files.length">点击或拖拽图片到此处上传</span>
    <span v-else>已选择 {{ files.length }} 张图片</span>
  </div>
  <input type="file" id="file" accept="image/*" multiple @change="handleFiles" ref="fileInput" style="display:none" />
  压缩后图片的质量（0~1）<input type="number" placeholder="请输入压缩后的质量（推荐0.8）" v-model.number="quality" min="0" max="1" step="0.01" />
  <div v-for="(img, idx) in previewList" :key="idx">
    <img :src="img" style="max-width: 200px;" />
  </div>
  <button @click="startCompression">点击我开始压缩</button>
  <button @click="downloadAll">点击我一键下载</button>
</div>

<script setup>

import { ref } from 'vue'
import Compressor from 'compressorjs'

const previewList = ref([])
const quality = ref(0.8)
const files = ref([])
const fileInput = ref(null)

function triggerFileInput() {
  fileInput.value && fileInput.value.click()
}

function handleFiles(e) {
  files.value = Array.from(e.target.files)
}

function handleDrop(e) {
  files.value = Array.from(e.dataTransfer.files)
}

function compressImage(file, quality) {
  return new Promise((resolve, reject) => {
    new Compressor(file, {
      quality: quality,
      success(result) {
        resolve(result)
      },
      error(err) {
        reject(err)
      },
    })
  })
}

function startCompression() {
  previewList.value = []
  files.value.forEach(async (file) => {
    try {
      const compressedFile = await compressImage(file, quality.value)
      const reader = new FileReader()
      reader.onload = (e) => {
        previewList.value.push(e.target.result)
      }
      reader.readAsDataURL(compressedFile)
    } catch (error) {
      console.error('Compression failed:', error)
    }
  })
}

function downloadAll() {
  files.value.forEach(async (file, index) => {
    try {
      const compressedFile = await compressImage(file, quality.value)
      const url = URL.createObjectURL(compressedFile)
      const a = document.createElement('a')
      a.href = url
      a.download = `compressed_${file.name}`
      document.body.appendChild(a)
      a.click()
      document.body.removeChild(a)
      URL.revokeObjectURL(url)
    } catch (error) {
      console.error('Download failed:', error)
    }
  })
}
</script>
