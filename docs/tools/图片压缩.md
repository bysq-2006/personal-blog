## 基于Compressor
[Compressor](https://github.com/fengyuanchen/compressorjs) 是一个基于 JavaScript 的图片压缩工具，支持在浏览器端对图片进行压缩处理。它可以有效地减小图片文件的大小，同时保持较高的图像质量，适用于各种 web 应用场景。

## 工具本身

<div class="main">
  <div class="upload-box" @click="triggerFileInput" @dragover.prevent @drop.prevent="handleDrop">
    <span v-if="!files.length">点击或拖拽图片到此处上传</span>
    <span v-else>已选择 {{ files.length }} 张图片</span>
  </div>
  <input type="file" id="file" accept="image/*" multiple @change="handleFiles" ref="fileInput" style="display:none" />
  
  <div class="controls">
    <div class="control-item">
      <label>压缩策略：</label>
      <select v-model="strategy">
        <option value="quality">算法压缩 (调整质量)</option>
        <option value="resize">基础缩放 (减少分辨率)</option>
      </select>
    </div>
    <div v-if="strategy === 'quality'" class="control-item">
      质量 (0~1): 
      <input type="number" v-model.number="quality" min="0" max="1" step="0.1" />
    </div>
    <div v-else class="control-item">
      缩放比例 (0~1): 
      <input type="number" v-model.number="scale" min="0.1" max="1" step="0.1" />
    </div>
  </div>

  <div class="preview-grid">
    <div v-for="(img, idx) in previewList" :key="idx" class="preview-card">
      <img :src="img.url" />
      <div class="img-info">
        <span>{{ img.name }}</span>
        <span>{{ (img.size / 1024).toFixed(2) }} KB</span>
      </div>
    </div>
  </div>
  <div class="actions">
    <button @click="startCompression">点击预览图片</button>
    <button @click="downloadAll" class="secondary">一键下载</button>
  </div>
</div>

<script setup>

import { ref } from 'vue'
import Compressor from 'compressorjs'

const previewList = ref([])
const quality = ref(0.8)
const scale = ref(0.5)
const strategy = ref('quality')
const files = ref([])
const fileInput = ref(null)

function triggerFileInput() {
  fileInput.value && fileInput.value.click()
}

function handleFiles(e) {
  files.value = Array.from(e.target.files)
}

function handleDrop(e) {
  files.value = Array.from(e.dataTransfer.files)
}

function compressImage(file, options) {
  return new Promise((resolve, reject) => {
    new Compressor(file, {
      ...options,
      success(result) {
        resolve(result)
      },
      error(err) {
        reject(err)
      },
    })
  })
}

async function getCompressedFile(file) {
  const options = {
    quality: strategy.value === 'quality' ? quality.value : 1,
  }

  if (strategy.value === 'resize') {
    // For resize, we need to calculate dimensions
    // CompressorJS can handle maxWidth/maxHeight, but for exact scale we might need to pre-calculate
    // Or just use its built-in scaling if we provide one dimension
    // Let's use a simpler approach: if resize, we set a very high quality but reduce size
    // To get the scale, we can use a helper to get image size
    const dimensions = await getImageDimensions(file)
    options.width = dimensions.width * scale.value
    options.height = dimensions.height * scale.value
  }

  return await compressImage(file, options)
}

function getImageDimensions(file) {
  return new Promise((resolve) => {
    const img = new Image()
    const url = URL.createObjectURL(file)
    img.onload = () => {
      resolve({ width: img.width, height: img.height })
      URL.revokeObjectURL(url)
    }
    img.src = url
  })
}

async function startCompression() {
  previewList.value = []
  for (const file of files.value) {
    try {
      const compressedFile = await getCompressedFile(file)
      const url = URL.createObjectURL(compressedFile)
      previewList.value.push({
        url,
        size: compressedFile.size,
        name: file.name
      })
    } catch (error) {
      console.error('Compression failed:', error)
    }
  }
}

async function downloadAll() {
  for (const file of files.value) {
    try {
      const compressedFile = await getCompressedFile(file)
      const url = URL.createObjectURL(compressedFile)
      const a = document.createElement('a')
      a.href = url
      a.download = `compressed_${file.name}`
      document.body.appendChild(a)
      a.click()
      document.body.removeChild(a)
      URL.revokeObjectURL(url)
    } catch (error) {
      console.error('Download failed:', error)
    }
  }
}
</script>
